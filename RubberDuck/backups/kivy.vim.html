1    " ~/.vim/after/syntax/kivy.vim
2    "
3    " Vim syntax file
4    " Language:	Kivy

7    
8    syntax clear
9    
10   syn include @pyth $VIMRUNTIME/syntax/python.vim
11   
12   syn match kivyComment       /#.*\n/ display contains=pythonTodo,Spell
13   syn match kivyPreProc       /^\s*#:.*/
14   syn match kivyRule          /<-*\I\i*\%([,@+]\I\i*\)*>:/
15   syn match kivyRule          /\[-*\I\i*\%([,@+]\I\i*\)*]:/
16   syn match kivyRootRule      /^\I\i*:\s*$/
17   
18   syn region kivyAttrBlock    matchgroup=kivyAttribute start=/^\z(\s\+\)\I\i*\s*:\s*$/ skip="^\s*$" end="^\%(\z1\s\{4}\)\@!" contains=@pyth
19   syn region kivyAttrBlock    matchgroup=kivyAttribute start=/^\s\+\I\i*\s*:\%(\s*\S\)\@=/ end="$" keepend oneline contains=@pyth
20   
21   syn region kivyId           matchgroup=kivyAttribute start=/^\s\+id\s*:\s*/ end="\w\+\zs" oneline
22   
23   syn region kivyBindBlock    matchgroup=kivyBind start=/^\z(\s\+\)on_\I\i*\s*:\s*$/ skip="^\s*$" end="^\%(\z1\s\{4}\)\@!" contains=@pyth
24   syn region kivyBindBlock    matchgroup=kivyBind start=/^\s\+on_\i\+\s*:\%(\s*\S\)\@=/ end="$" keepend oneline contains=@pyth
25   
26   syn region kivyCanvasValue  matchgroup=kivyCanvas start=/^\z(\s\+\)\I\i*\s*:\s*$/ skip="^\s*$" end="^\%(\z1\s\{4}\)\@!" contains=@pyth contained
27   syn region kivyCanvasValue  matchgroup=kivyCanvas start=/^\s\+\I\i*\s*:\%(\s*\S\)\@=/ end="$" keepend oneline contains=@pyth contained
28   syn region kivyCanvas       matchgroup=kivyCanvas start=/^\z(\s\+\)canvas.*:\s*$/ skip="^\s*$" end="^\%(\z1\s\{4}\)\@!"
29                               \   contains=kivyInstruction,kivyPreProc,kivyComment,kivyCanvasValue 
30   
31   syn match kivyInstruction   /^\s\+\u\i*\s*:/ contained
32   syn match kivyWidget        /^\s\+\u\i*\s*:/
33   
34   hi def link kivyPreproc     PreProc
35   hi def link kivyComment     Comment
36   hi def link kivyRule        Type
37   hi def link kivyRootRule    Function
38   hi def link kivyAttribute   Label
39   hi def link kivyBind        Function
40   hi def link kivyWidget      Function
41   hi def link kivyCanvas      special
42   hi def link kivyInstruction Statement
43   
44   hi KivyId cterm=underline
45   hi KivyPreproc cterm=bold
46   
47   let b:current_syntax = "kivy"
48
