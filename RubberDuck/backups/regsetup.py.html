11   def FileExists(fname):
12       """Check if a file exists.  Returns true or false."""
13       import os
14   
15       try:
16           os.stat(fname)
17           return 1
18       except os.error as details:
19           return 0

22   def IsPackageDir(path, packageName, knownFileName):
23       """Given a path, a ni package name, and possibly a known file name in 
24       the root of the package, see if this path is good. 
25       """
26       import os
27   
28       if knownFileName is None:
29           knownFileName = "."
30       return FileExists(os.path.join(os.path.join(path, packageName), knownFileName))

34       """Return "_d" if we're running a debug version. 
35    
36       This is to be used within DLL names when locating them. 
37       """
38       import importlib.machinery
39   
40       return "_d" if "_d.pyd" in importlib.machinery.EXTENSION_SUFFIXES else ""

43   def FindPackagePath(packageName, knownFileName, searchPaths):
44       """Find a package. 
45    
46       Given a ni style package name, check the package is registered. 
47    
48       First place looked is the registry for an existing entry.  Then 
49       the searchPaths are searched. 
50       """
51       import regutil, os
52   
53       pathLook = regutil.GetRegisteredNamedPath(packageName)
54       if pathLook and IsPackageDir(pathLook, packageName, knownFileName):
55           return pathLook, None  # The currently registered one is good.
56       # Search down the search paths.
57       for pathLook in searchPaths:
58           if IsPackageDir(pathLook, packageName, knownFileName):
59               # Found it
60               ret = os.path.abspath(pathLook)
61               return ret, ret
62       raise error("The package %s can not be located" % packageName)

65   def FindHelpPath(helpFile, helpDesc, searchPaths):
66       # See if the current registry entry is OK
67       import os, win32api, win32con
68   
69       try:
70           key = win32api.RegOpenKey(
71               win32con.HKEY_LOCAL_MACHINE,
72               "Software\\Microsoft\\Windows\\Help",
73               0,
74               win32con.KEY_ALL_ACCESS,
75           )
76           try:
77               try:
78                   path = win32api.RegQueryValueEx(key, helpDesc)[0]
79                   if FileExists(os.path.join(path, helpFile)):
80                       return os.path.abspath(path)
81               except win32api.error:
82                   pass  # no registry entry.
83           finally:
84               key.Close()
85       except win32api.error:
86           pass
87       for pathLook in searchPaths:
88           if FileExists(os.path.join(pathLook, helpFile)):
89               return os.path.abspath(pathLook)
90           pathLook = os.path.join(pathLook, "Help")
91           if FileExists(os.path.join(pathLook, helpFile)):
92               return os.path.abspath(pathLook)
93       raise error("The help file %s can not be located" % helpFile)

96   def FindAppPath(appName, knownFileName, searchPaths):
97       """Find an application. 
98    
99       First place looked is the registry for an existing entry.  Then 
100      the searchPaths are searched. 
101      """
102      # Look in the first path.
103      import regutil, string, os
104  
105      regPath = regutil.GetRegisteredNamedPath(appName)
106      if regPath:
107          pathLook = regPath.split(";")[0]
108      if regPath and FileExists(os.path.join(pathLook, knownFileName)):
109          return None  # The currently registered one is good.
110      # Search down the search paths.
111      for pathLook in searchPaths:
112          if FileExists(os.path.join(pathLook, knownFileName)):
113              # Found it
114              return os.path.abspath(pathLook)
115      raise error(
116          "The file %s can not be located for application %s" % (knownFileName, appName)
117      )

120  def FindPythonExe(exeAlias, possibleRealNames, searchPaths):
121      """Find an exe. 
122   
123      Returns the full path to the .exe, and a boolean indicating if the current 
124      registered entry is OK.  We don't trust the already registered version even 
125      if it exists - it may be wrong (ie, for a different Python version) 
126      """
127      import win32api, regutil, string, os, sys
128  
129      if possibleRealNames is None:
130          possibleRealNames = exeAlias
131      # Look first in Python's home.
132      found = os.path.join(sys.prefix, possibleRealNames)
133      if not FileExists(found):  # for developers
134          if "64 bit" in sys.version:
135              found = os.path.join(sys.prefix, "PCBuild", "amd64", possibleRealNames)
136          else:
137              found = os.path.join(sys.prefix, "PCBuild", possibleRealNames)
138      if not FileExists(found):
139          found = LocateFileName(possibleRealNames, searchPaths)
140  
141      registered_ok = 0
142      try:
143          registered = win32api.RegQueryValue(
144              regutil.GetRootKey(), regutil.GetAppPathsKey() + "\\" + exeAlias
145          )
146          registered_ok = found == registered
147      except win32api.error:
148          pass
149      return found, registered_ok

152  def QuotedFileName(fname):
153      """Given a filename, return a quoted version if necessary"""
154      import regutil, string
155  
156      try:
157          fname.index(" ")  # Other chars forcing quote?
158          return '"%s"' % fname
159      except ValueError:
160          # No space in name.
161          return fname

164  def LocateFileName(fileNamesString, searchPaths):
165      """Locate a file name, anywhere on the search path. 
166   
167      If the file can not be located, prompt the user to find it for us 
168      (using a common OpenFile dialog) 
169   
170      Raises KeyboardInterrupt if the user cancels. 
171      """
172      import regutil, string, os
173  
174      fileNames = fileNamesString.split(";")
175      for path in searchPaths:
176          for fileName in fileNames:
177              try:
178                  retPath = os.path.join(path, fileName)
179                  os.stat(retPath)
180                  break
181              except os.error:
182                  retPath = None
183          if retPath:
184              break
185      else:
186          fileName = fileNames[0]
187          try:
188              import win32ui, win32con
189          except ImportError:
190              raise error(
191                  "Need to locate the file %s, but the win32ui module is not available\nPlease run the program again, passing as a parameter the path to this file."
192                  % fileName
193              )
194          # Display a common dialog to locate the file.
195          flags = win32con.OFN_FILEMUSTEXIST
196          ext = os.path.splitext(fileName)[1]
197          filter = "Files of requested type (*%s)|*%s||" % (ext, ext)
198          dlg = win32ui.CreateFileDialog(1, None, fileName, flags, filter, None)
199          dlg.SetOFNTitle("Locate " + fileName)
200          if dlg.DoModal() != win32con.IDOK:
201              raise KeyboardInterrupt("User cancelled the process")
202          retPath = dlg.GetPathName()
203      return os.path.abspath(retPath)

229  def LocatePythonCore(searchPaths):
230      """Locate and validate the core Python directories.  Returns a list 
231      of paths that should be used as the core (ie, un-named) portion of 
232      the Python path. 
233      """
234      import os, regutil
235  
236      currentPath = regutil.GetRegisteredNamedPath(None)
237      if currentPath:
238          presearchPaths = currentPath.split(";")
239      else:
240          presearchPaths = [os.path.abspath(".")]
241      libPath = None
242      for path in presearchPaths:
243          if FileExists(os.path.join(path, "os.py")):
244              libPath = path
245              break
246      if libPath is None and searchPaths is not None:
247          libPath = LocatePath("os.py", searchPaths)
248      if libPath is None:
249          raise error("The core Python library could not be located.")
250  
251      corePath = None
252      suffix = IsDebug()
253      for path in presearchPaths:
254          if FileExists(os.path.join(path, "unicodedata%s.pyd" % suffix)):
255              corePath = path
256              break
257      if corePath is None and searchPaths is not None:
258          corePath = LocatePath("unicodedata%s.pyd" % suffix, searchPaths)
259      if corePath is None:
260          raise error("The core Python path could not be located.")
261  
262      installPath = os.path.abspath(os.path.join(libPath, ".."))
263      return installPath, [libPath, corePath]

266  def FindRegisterPackage(packageName, knownFile, searchPaths, registryAppName=None):
267      """Find and Register a package. 
268   
269      Assumes the core registry setup correctly. 
270   
271      In addition, if the location located by the package is already 
272      in the **core** path, then an entry is registered, but no path. 
273      (no other paths are checked, as the application whose path was used 
274      may later be uninstalled.  This should not happen with the core) 
275      """
276      import regutil, string
277  
278      if not packageName:
279          raise error("A package name must be supplied")
280      corePaths = regutil.GetRegisteredNamedPath(None).split(";")
281      if not searchPaths:
282          searchPaths = corePaths
283      registryAppName = registryAppName or packageName
284      try:
285          pathLook, pathAdd = FindPackagePath(packageName, knownFile, searchPaths)
286          if pathAdd is not None:
287              if pathAdd in corePaths:
288                  pathAdd = ""
289              regutil.RegisterNamedPath(registryAppName, pathAdd)
290          return pathLook
291      except error as details:
292          print(
293              "*** The %s package could not be registered - %s" % (packageName, details)
294          )
295          print(
296              "*** Please ensure you have passed the correct paths on the command line."
297          )
298          print(
299              "*** - For packages, you should pass a path to the packages parent directory,"
300          )
301          print("*** - and not the package directory itself...")

304  def FindRegisterApp(appName, knownFiles, searchPaths):
305      """Find and Register a package. 
306   
307      Assumes the core registry setup correctly. 
308   
309      """
310      import regutil, string
311  
312      if type(knownFiles) == type(""):
313          knownFiles = [knownFiles]
314      paths = []
315      try:
316          for knownFile in knownFiles:
317              pathLook = FindAppPath(appName, knownFile, searchPaths)
318              if pathLook:
319                  paths.append(pathLook)
320      except error as details:
321          print("*** ", details)
322          return
323  
324      regutil.RegisterNamedPath(appName, ";".join(paths))

352  def SetupCore(searchPaths):
353      """Setup the core Python information in the registry. 
354   
355      This function makes no assumptions about the current state of sys.path. 
356   
357      After this function has completed, you should have access to the standard 
358      Python library, and the standard Win32 extensions 
359      """
360  
361      import sys
362  
363      for path in searchPaths:
364          sys.path.append(path)
365  
366      import os
367      import regutil, win32api, win32con
368  
369      installPath, corePaths = LocatePythonCore(searchPaths)
370      # Register the core Pythonpath.
371      print(corePaths)
372      regutil.RegisterNamedPath(None, ";".join(corePaths))
373  
374      # Register the install path.
375      hKey = win32api.RegCreateKey(regutil.GetRootKey(), regutil.BuildDefaultPythonKey())
376      try:
377          # Core Paths.
378          win32api.RegSetValue(hKey, "InstallPath", win32con.REG_SZ, installPath)
379      finally:
380          win32api.RegCloseKey(hKey)
381  
382      # Register the win32 core paths.
383      win32paths = (
384          os.path.abspath(os.path.split(win32api.__file__)[0])
385          + ";"
386          + os.path.abspath(
387              os.path.split(LocateFileName("win32con.py;win32con.pyc", sys.path))[0]
388          )
389      )
390  
391      # Python has builtin support for finding a "DLLs" directory, but
392      # not a PCBuild.  Having it in the core paths means it is ignored when
393      # an EXE not in the Python dir is hosting us - so we add it as a named
394      # value
395      check = os.path.join(sys.prefix, "PCBuild")
396      if "64 bit" in sys.version:
397          check = os.path.join(check, "amd64")
398      if os.path.isdir(check):
399          regutil.RegisterNamedPath("PCBuild", check)

402  def RegisterShellInfo(searchPaths):
403      """Registers key parts of the Python installation with the Windows Shell. 
404   
405      Assumes a valid, minimal Python installation exists 
406      (ie, SetupCore() has been previously successfully run) 
407      """
408      import regutil, win32con
409  
410      suffix = IsDebug()
411      # Set up a pointer to the .exe's
412      exePath = FindRegisterPythonExe("Python%s.exe" % suffix, searchPaths)
413      regutil.SetRegistryDefaultValue(".py", "Python.File", win32con.HKEY_CLASSES_ROOT)
414      regutil.RegisterShellCommand("Open", QuotedFileName(exePath) + ' "%1" %*', "&Run")
415      regutil.SetRegistryDefaultValue(
416          "Python.File\\DefaultIcon", "%s,0" % exePath, win32con.HKEY_CLASSES_ROOT
417      )
418  
419      FindRegisterHelpFile("Python.hlp", searchPaths, "Main Python Documentation")
420      FindRegisterHelpFile("ActivePython.chm", searchPaths, "Main Python Documentation")
421  
422      # We consider the win32 core, as it contains all the win32 api type
423      # stuff we need.
424

428  usage = (
429      """\ 
430  regsetup.py - Setup/maintain the registry for Python apps. 
431   
432  Run without options, (but possibly search paths) to repair a totally broken 
433  python registry setup.  This should allow other options to work. 
434   
435  Usage:   %s [options ...] paths ... 
436  -p packageName  -- Find and register a package.  Looks in the paths for 
437                     a sub-directory with the name of the package, and 
438                     adds a path entry for the package. 
439  -a appName      -- Unconditionally add an application name to the path. 
440                     A new path entry is create with the app name, and the 
441                     paths specified are added to the registry. 
442  -c              -- Add the specified paths to the core Pythonpath. 
443                     If a path appears on the core path, and a package also 
444                     needs that same path, the package will not bother 
445                     registering it.  Therefore, By adding paths to the 
446                     core path, you can avoid packages re-registering the same path. 
447  -m filename     -- Find and register the specific file name as a module. 
448                     Do not include a path on the filename! 
449  --shell         -- Register everything with the Win95/NT shell. 
450  --upackage name -- Unregister the package 
451  --uapp name     -- Unregister the app (identical to --upackage) 
452  --umodule name  -- Unregister the module 
453   
454  --description   -- Print a description of the usage. 
455  --examples      -- Print examples of usage. 
456  """
457      % sys.argv[0]
458  )
459  
460  description = """\ 
461  If no options are processed, the program attempts to validate and set 
462  the standard Python path to the point where the standard library is 
463  available.  This can be handy if you move Python to a new drive/sub-directory, 
464  in which case most of the options would fail (as they need at least string.py, 
465  os.py etc to function.) 
466  Running without options should repair Python well enough to run with 
467  the other options. 
468   
469  paths are search paths that the program will use to seek out a file. 
470  For example, when registering the core Python, you may wish to 
471  provide paths to non-standard places to look for the Python help files, 
472  library files, etc. 
473   
474  See also the "regcheck.py" utility which will check and dump the contents 
475  of the registry. 
476  """
477  
478  examples = """\ 
479  Examples: 
480  "regsetup c:\\wierd\\spot\\1 c:\\wierd\\spot\\2" 
481  Attempts to setup the core Python.  Looks in some standard places, 
482  as well as the 2 wierd spots to locate the core Python files (eg, Python.exe, 
483  python14.dll, the standard library and Win32 Extensions. 
484   
485  "regsetup -a myappname . .\subdir" 
486  Registers a new Pythonpath entry named myappname, with "C:\\I\\AM\\HERE" and 
487  "C:\\I\\AM\\HERE\subdir" added to the path (ie, all args are converted to 
488  absolute paths) 
489   
490  "regsetup -c c:\\my\\python\\files" 
491  Unconditionally add "c:\\my\\python\\files" to the 'core' Python path. 
492   
493  "regsetup -m some.pyd \\windows\\system" 
494  Register the module some.pyd in \\windows\\system as a registered 
495  module.  This will allow some.pyd to be imported, even though the 
496  windows system directory is not (usually!) on the Python Path. 
497   
498  "regsetup --umodule some" 
499  Unregister the module "some".  This means normal import rules then apply 
500  for that module. 
501  """
502  
503  if __name__ == "__main__":
504      if len(sys.argv) > 1 and sys.argv[1] in ["/?", "-?", "-help", "-h"]:
505          print(usage)
506      elif len(sys.argv) == 1 or not sys.argv[1][0] in ["/", "-"]:
507          # No args, or useful args.
508          searchPath = sys.path[:]
509          for arg in sys.argv[1:]:
510              searchPath.append(arg)
511          # Good chance we are being run from the "regsetup.py" directory.
512          # Typically this will be "\somewhere\win32\Scripts" and the
513          # "somewhere" and "..\Lib" should also be searched.
514          searchPath.append("..\\Build")
515          searchPath.append("..\\Lib")
516          searchPath.append("..")
517          searchPath.append("..\\..")
518  
519          # for developers:
520          # also search somewhere\lib, ..\build, and ..\..\build
521          searchPath.append("..\\..\\lib")
522          searchPath.append("..\\build")
523          if "64 bit" in sys.version:
524              searchPath.append("..\\..\\pcbuild\\amd64")
525          else:
526              searchPath.append("..\\..\\pcbuild")
527  
528          print("Attempting to setup/repair the Python core")
529  
530          SetupCore(searchPath)
531          RegisterShellInfo(searchPath)
532          FindRegisterHelpFile("PyWin32.chm", searchPath, "Pythonwin Reference")
533          # Check the registry.
534          print("Registration complete - checking the registry...")
535          import regcheck
536  
537          regcheck.CheckRegistry()
538      else:
539          searchPaths = []
540          import getopt, string
541  
542          opts, args = getopt.getopt(
543              sys.argv[1:],
544              "p:a:m:c",
545              ["shell", "upackage=", "uapp=", "umodule=", "description", "examples"],
546          )
547          for arg in args:
548              searchPaths.append(arg)
549          for o, a in opts:
550              if o == "--description":
551                  print(description)
552              if o == "--examples":
553                  print(examples)
554              if o == "--shell":
555                  print("Registering the Python core.")
556                  RegisterShellInfo(searchPaths)
557              if o == "-p":
558                  print("Registering package", a)
559                  FindRegisterPackage(a, None, searchPaths)
560              if o in ["--upackage", "--uapp"]:
561                  import regutil
562  
563                  print("Unregistering application/package", a)
564                  regutil.UnregisterNamedPath(a)
565              if o == "-a":
566                  import regutil
567  
568                  path = ";".join(searchPaths)
569                  print("Registering application", a, "to path", path)
570                  regutil.RegisterNamedPath(a, path)
571              if o == "-c":
572                  if not len(searchPaths):
573                      raise error("-c option must provide at least one additional path")
574                  import win32api, regutil
575  
576                  currentPaths = regutil.GetRegisteredNamedPath(None).split(";")
577                  oldLen = len(currentPaths)
578                  for newPath in searchPaths:
579                      if newPath not in currentPaths:
580                          currentPaths.append(newPath)
581                  if len(currentPaths) != oldLen:
582                      print(
583                          "Registering %d new core paths" % (len(currentPaths) - oldLen)
584                      )
585                      regutil.RegisterNamedPath(None, ";".join(currentPaths))
586                  else:
587                      print("All specified paths are already registered.")
588
