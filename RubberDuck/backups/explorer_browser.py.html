1    # A sample of using Vista's IExplorerBrowser interfaces...
2    # Currently doesn't quite work:

5    import sys
6    import pythoncom
7    from win32com.shell import shell, shellcon
8    import win32gui, win32con, win32api
9    from win32com.server.util import wrap, unwrap
10   
11   # event handler for the browser.
12   IExplorerBrowserEvents_Methods = """OnNavigationComplete OnNavigationFailed  
13                                       OnNavigationPending OnViewCreated""".split()
14   
15   
16   class EventHandler:
17       _com_interfaces_ = [shell.IID_IExplorerBrowserEvents]
18       _public_methods_ = IExplorerBrowserEvents_Methods
19   
20       def OnNavigationComplete(self, pidl):
21           print("OnNavComplete", pidl)
22   
23       def OnNavigationFailed(self, pidl):
24           print("OnNavigationFailed", pidl)
25   
26       def OnNavigationPending(self, pidl):
27           print("OnNavigationPending", pidl)
28   
29       def OnViewCreated(self, view):
30           print("OnViewCreated", view)
31           # And if our demo view has been registered, it may well
32           # be that view!
33           try:
34               pyview = unwrap(view)
35               print("and look - its a Python implemented view!", pyview)
36           except ValueError:
37               pass
38   
39   
40   class MainWindow:
41       def __init__(self):
42           message_map = {
43               win32con.WM_DESTROY: self.OnDestroy,
44               win32con.WM_COMMAND: self.OnCommand,
45               win32con.WM_SIZE: self.OnSize,
46           }
47           # Register the Window class.
48           wc = win32gui.WNDCLASS()
49           hinst = wc.hInstance = win32api.GetModuleHandle(None)
50           wc.lpszClassName = "test_explorer_browser"
51           wc.lpfnWndProc = message_map  # could also specify a wndproc.
52           classAtom = win32gui.RegisterClass(wc)
53           # Create the Window.
54           style = win32con.WS_OVERLAPPEDWINDOW | win32con.WS_VISIBLE
55           self.hwnd = win32gui.CreateWindow(
56               classAtom,
57               "Python IExplorerBrowser demo",
58               style,
59               0,
60               0,
61               win32con.CW_USEDEFAULT,
62               win32con.CW_USEDEFAULT,
63               0,
64               0,
65               hinst,
66               None,
67           )
68           eb = pythoncom.CoCreateInstance(
69               shellcon.CLSID_ExplorerBrowser,
70               None,
71               pythoncom.CLSCTX_ALL,
72               shell.IID_IExplorerBrowser,
73           )
74           # as per MSDN docs, hook up events early
75           self.event_cookie = eb.Advise(wrap(EventHandler()))
76   
77           eb.SetOptions(shellcon.EBO_SHOWFRAMES)
78           rect = win32gui.GetClientRect(self.hwnd)
79           # Set the flags such that the folders autoarrange and non web view is presented
80           flags = (shellcon.FVM_LIST, shellcon.FWF_AUTOARRANGE | shellcon.FWF_NOWEBVIEW)
81           eb.Initialize(self.hwnd, rect, (0, shellcon.FVM_DETAILS))
82           if len(sys.argv) == 2:
83               # If an arg was specified, ask the desktop parse it.
84               # You can pass anything explorer accepts as its '/e' argument -
85               # eg, "::{guid}\::{guid}" etc.
86               # "::{20D04FE0-3AEA-1069-A2D8-08002B30309D}" is "My Computer"
87               pidl = shell.SHGetDesktopFolder().ParseDisplayName(0, None, sys.argv[1])[1]
88           else:
89               # And start browsing at the root of the namespace.
90               pidl = []
91           eb.BrowseToIDList(pidl, shellcon.SBSP_ABSOLUTE)
92           # and for some reason the "Folder" view in the navigator pane doesn't
93           # magically synchronize itself - so let's do that ourself.
94           # Get the tree control.
95           sp = eb.QueryInterface(pythoncom.IID_IServiceProvider)
96           try:
97               tree = sp.QueryService(
98                   shell.IID_INameSpaceTreeControl, shell.IID_INameSpaceTreeControl
99               )
100          except pythoncom.com_error as exc:
101              # this should really only fail if no "nav" frame exists...
102              print(
103                  "Strange - failed to get the tree control even though "
104                  "we asked for a EBO_SHOWFRAMES"
105              )
106              print(exc)
107          else:
108              # get the IShellItem for the selection.
109              si = shell.SHCreateItemFromIDList(pidl, shell.IID_IShellItem)
110              # set it to selected.
111              tree.SetItemState(si, shellcon.NSTCIS_SELECTED, shellcon.NSTCIS_SELECTED)
112  
113          # eb.FillFromObject(None, shellcon.EBF_NODROPTARGET);
114          # eb.SetEmptyText("No known folders yet...");
115          self.eb = eb
116  
117      def OnCommand(self, hwnd, msg, wparam, lparam):
118          pass
119  
120      def OnDestroy(self, hwnd, msg, wparam, lparam):
121          print("tearing down ExplorerBrowser...")
122          self.eb.Unadvise(self.event_cookie)
123          self.eb.Destroy()
124          self.eb = None
125          print("shutting down app...")
126          win32gui.PostQuitMessage(0)
127  
128      def OnSize(self, hwnd, msg, wparam, lparam):
129          x = win32api.LOWORD(lparam)
130          y = win32api.HIWORD(lparam)
131          self.eb.SetRect(None, (0, 0, x, y))
132

134  def main():
135      w = MainWindow()
136      win32gui.PumpMessages()

138  
139  if __name__ == "__main__":
140      main()
