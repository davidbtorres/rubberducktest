27   def demo():
28       try:
29           # seeif I can locate the demo files.
30           import fontdemo
31       except ImportError:
32           # else put the demos direectory on the path (if not already)
33           try:
34               instPath = regutil.GetRegistryDefaultValue(
35                   regutil.BuildDefaultPythonKey() + "\\InstallPath"
36               )
37           except win32api.error:
38               print(
39                   "The InstallPath can not be located, and the Demos directory is not on the path"
40               )
41               instPath = "."
42   
43           demosDir = win32ui.FullPath(instPath + "\\Demos")
44           for path in sys.path:
45               if win32ui.FullPath(path) == demosDir:
46                   break
47           else:
48               sys.path.append(demosDir)
49           import fontdemo
50   
51       import sys
52   
53       if "/go" in sys.argv:
54           for name, cmd in demos:
55               try:
56                   exec(cmd)
57               except:
58                   print(
59                       "Demo of %s failed - %s:%s"
60                       % (cmd, sys.exc_info()[0], sys.exc_info()[1])
61                   )
62           return
63       # Otherwise allow the user to select the demo to run
64   
65       import pywin.dialogs.list
66   
67       while 1:
68           rc = pywin.dialogs.list.SelectFromLists("Select a Demo", demos, ["Demo Title"])
69           if rc is None:
70               break
71           title, cmd = demos[rc]
72           try:
73               exec(cmd)
74           except:
75               print(
76                   "Demo of %s failed - %s:%s"
77                   % (title, sys.exc_info()[0], sys.exc_info()[1])
78               )
