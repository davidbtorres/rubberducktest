1    ''' 
2    This is a extended unittest module for Kivy, to make unittests based on 
3    graphics with an OpenGL context. 
4     
5    The idea is to render a Widget tree, and after 1, 2 or more frames, a 
6    screenshot will be made and be compared to the original one. 
7    If no screenshot exists for the current test, the very first one will be used. 
8     
9    The screenshots live in the 'kivy/tests/results' folder and are in PNG format, 
10   320x240 pixels. 
11   '''

48   def ensure_web_server(root=None):
49       if http_server is not None:
50           return True
51   
52       if not root:
53           root = os.path.join(os.path.dirname(__file__), "..", "..")
54       need_chdir = sys.version_info.major == 3 and sys.version_info.minor <= 6
55       curr_dir = os.getcwd()
56   
57       def _start_web_server():
58           global http_server
59           from http.server import SimpleHTTPRequestHandler
60           from socketserver import TCPServer
61   
62           try:
63               if need_chdir:
64                   os.chdir(root)
65                   handler = SimpleHTTPRequestHandler
66               else:
67                   handler = partial(SimpleHTTPRequestHandler, directory=root)
68   
69               http_server = TCPServer(
70                   ("", 8000), handler, bind_and_activate=False)
71               http_server.daemon_threads = True
72               http_server.allow_reuse_address = True
73               http_server.server_bind()
74               http_server.server_activate()
75               http_server_ready.set()
76               http_server.serve_forever()
77           except:
78               import traceback
79               traceback.print_exc()
80           finally:
81               http_server = None
82               http_server_ready.set()
83   
84               if need_chdir:
85                   os.chdir(curr_dir)
86   
87       th = threading.Thread(target=_start_web_server)
88       th.daemon = True
89       th.start()
90       http_server_ready.wait()
91       if http_server is None:
92           raise Exception("Unable to start webserver")

95   class GraphicUnitTest(_base):
96       framecount = 0
97   
98       def _force_refresh(self, *largs):
99           # this prevent in some case to be stuck if the screen doesn't refresh
100          # and we wait for a number of self.framecount that never goes down
101          from kivy.base import EventLoop
102          win = EventLoop.window
103          if win and win.canvas:
104              win.canvas.ask_update()
105  
106      def render(self, root, framecount=1):
107          '''Call rendering process using the `root` widget. 
108          The screenshot will be done in `framecount` frames. 
109          '''
110          from kivy.base import runTouchApp
111          from kivy.clock import Clock
112          self.framecount = framecount
113          try:
114              Clock.schedule_interval(self._force_refresh, 1)
115              runTouchApp(root)
116          finally:
117              Clock.unschedule(self._force_refresh)
118  
119          # reset for the next test, but nobody will know if it will be used :/
120          if self.test_counter != 0:
121              self.tearDown(fake=True)
122              self.setUp()
123  
124      def run(self, *args, **kwargs):
125          '''Extend the run of unittest, to check if results directory have been 
126          found. If no results directory exists, the test will be ignored. 
127          '''
128          from os.path import join, dirname, exists
129          results_dir = join(dirname(__file__), 'results')
130          if make_screenshots and not exists(results_dir):
131              log.warning('No result directory found, cancel test.')
132              os.mkdir(results_dir)
133          self.test_counter = 0
134          self.results_dir = results_dir
135          self.test_failed = False
136          return super(GraphicUnitTest, self).run(*args, **kwargs)

138      def setUp(self):
139          '''Prepare the graphic test, with: 
140              - Window size fixed to 320x240 
141              - Default kivy configuration 
142              - Without any kivy input 
143          '''
144  
145          # use default kivy configuration (don't load user file.)
146          from os import environ
147          environ['KIVY_USE_DEFAULTCONFIG'] = '1'
148  
149          # force window size + remove all inputs
150          from kivy.config import Config
151          Config.set('graphics', 'width', '320')
152          Config.set('graphics', 'height', '240')
153          for items in Config.items('input'):
154              Config.remove_option('input', items[0])
155  
156          # bind ourself for the later screenshot
157          from kivy.core.window import Window
158          self.Window = Window
159          Window.bind(on_flip=self.on_window_flip)
160  
161          # ensure our window is correctly created
162          Window.create_window()
163          Window.register()
164          Window.initialized = True
165          Window.canvas.clear()
166          Window.close = lambda *s: True

168      def on_window_flip(self, window):
169          '''Internal method to be called when the window have just displayed an 
170          image. 
171          When an image is showed, we decrement our framecount. If framecount is 
172          come to 0, we are taking the screenshot. 
173   
174          The screenshot is done in a temporary place, and is compared to the 
175          original one -> test ok/ko. 
176          If no screenshot is available in the results directory, a new one will 
177          be created. 
178          '''
179          from kivy.base import EventLoop
180          from tempfile import mkstemp
181          from os.path import join, exists
182          from os import unlink, close
183          from shutil import move, copy
184  
185          # don't save screenshot until we have enough frames.
186          # log.debug('framecount %d' % self.framecount)
187          # ! check if there is 'framecount', otherwise just
188          # ! assume zero e.g. if handling runTouchApp manually
189          self.framecount = getattr(self, 'framecount', 0) - 1
190          if self.framecount > 0:
191              return
192  
193          # don't create screenshots if not requested manually
194          if not make_screenshots:
195              EventLoop.stop()
196              return
197  
198          reffn = None
199          match = False
200          try:
201              # just get a temporary name
202              fd, tmpfn = mkstemp(suffix='.png', prefix='kivyunit-')
203              close(fd)
204              unlink(tmpfn)
205  
206              # get a filename for the current unit test
207              self.test_counter += 1
208              test_uid = '%s-%d.png' % (
209                  '_'.join(self.id().split('.')[-2:]),
210                  self.test_counter)
211  
212              # capture the screen
213              log.info('Capturing screenshot for %s' % test_uid)
214              tmpfn = window.screenshot(tmpfn)
215              log.info('Capture saved at %s' % tmpfn)
216  
217              # search the file to compare to
218              reffn = join(self.results_dir, test_uid)
219              log.info('Compare with %s' % reffn)
220  
221              # get sourcecode
222              import inspect
223              frame = inspect.getouterframes(inspect.currentframe())[6]
224              sourcecodetab, line = inspect.getsourcelines(frame[0])
225              line = frame[2] - line
226              currentline = sourcecodetab[line]
227              sourcecodetab[line] = '<span style="color: red;">%s</span>' % (
228                  currentline)
229              sourcecode = ''.join(sourcecodetab)
230              sourcecodetab[line] = '>>>>>>>>\n%s<<<<<<<<\n' % currentline
231              sourcecodeask = ''.join(sourcecodetab)
232  
233              if not exists(reffn):
234                  log.info('No image reference, move %s as ref ?' % test_uid)
235                  if self.interactive_ask_ref(sourcecodeask, tmpfn, self.id()):
236                      move(tmpfn, reffn)
237                      tmpfn = reffn
238                      log.info('Image used as reference')
239                      match = True
240                  else:
241                      log.info('Image discarded')
242              else:
243                  from kivy.core.image import Image as CoreImage
244                  s1 = CoreImage(tmpfn, keep_data=True)
245                  sd1 = s1.image._data[0].data
246                  s2 = CoreImage(reffn, keep_data=True)
247                  sd2 = s2.image._data[0].data
248                  if sd1 != sd2:
249                      log.critical(
250                          '%s at render() #%d, images are different.' % (
251                              self.id(), self.test_counter))
252                      if self.interactive_ask_diff(sourcecodeask,
253                                                   tmpfn, reffn, self.id()):
254                          log.critical('user ask to use it as ref.')
255                          move(tmpfn, reffn)
256                          tmpfn = reffn
257                          match = True
258                      else:
259                          self.test_failed = True
260                  else:
261                      match = True
262  
263              # generate html
264              from os.path import join, dirname, exists, basename
265              from os import mkdir
266              build_dir = join(dirname(__file__), 'build')
267              if not exists(build_dir):
268                  mkdir(build_dir)
269              copy(reffn, join(build_dir, 'ref_%s' % basename(reffn)))
270              if tmpfn != reffn:
271                  copy(tmpfn, join(build_dir, 'test_%s' % basename(reffn)))
272              with open(join(build_dir, 'index.html'), 'at') as fd:
273                  color = '#ffdddd' if not match else '#ffffff'
274                  fd.write('<div style="background-color: %s">' % color)
275                  fd.write('<h2>%s #%d</h2>' % (self.id(), self.test_counter))
276                  fd.write('<table><tr><th>Reference</th>'
277                           '<th>Test</th>'
278                           '<th>Comment</th>')
279                  fd.write('<tr><td><img src="ref_%s"/></td>' %
280                           basename(reffn))
281                  if tmpfn != reffn:
282                      fd.write('<td><img src="test_%s"/></td>' %
283                               basename(reffn))
284                  else:
285                      fd.write('<td>First time, no comparison.</td>')
286                  fd.write('<td><pre>%s</pre></td>' % sourcecode)
287                  fd.write('</table></div>')
288          finally:
289              try:
290                  if reffn != tmpfn:
291                      unlink(tmpfn)
292              except:
293                  pass
294              EventLoop.stop()

296      def tearDown(self, fake=False):
297          '''When the test is finished, stop the application, and unbind our 
298          current flip callback. 
299          '''
300          from kivy.base import stopTouchApp
301          from kivy.core.window import Window
302          from kivy.clock import Clock
303          Window.unbind(on_flip=self.on_window_flip)
304          stopTouchApp()
305  
306          if not fake and self.test_failed:
307              self.assertTrue(False)
308          super(GraphicUnitTest, self).tearDown()

310      def interactive_ask_ref(self, code, imagefn, testid):
311          from os import environ
312          if 'UNITTEST_INTERACTIVE' not in environ:
313              return True
314  
315          from tkinter import Tk, Label, LEFT, RIGHT, BOTTOM, Button
316          from PIL import Image, ImageTk
317  
318          self.retval = False
319  
320          root = Tk()
321  
322          def do_close():
323              root.destroy()
324  
325          def do_yes():
326              self.retval = True
327              do_close()
328  
329          image = Image.open(imagefn)
330          photo = ImageTk.PhotoImage(image)
331          Label(root, text='The test %s\nhave no reference.' % testid).pack()
332          Label(root, text='Use this image as a reference ?').pack()
333          Label(root, text=code, justify=LEFT).pack(side=RIGHT)
334          Label(root, image=photo).pack(side=LEFT)
335          Button(root, text='Use as reference', command=do_yes).pack(side=BOTTOM)
336          Button(root, text='Discard', command=do_close).pack(side=BOTTOM)
337          root.mainloop()
338  
339          return self.retval

341      def interactive_ask_diff(self, code, tmpfn, reffn, testid):
342          from os import environ
343          if 'UNITTEST_INTERACTIVE' not in environ:
344              return False
345  
346          from tkinter import Tk, Label, LEFT, RIGHT, BOTTOM, Button
347          from PIL import Image, ImageTk
348  
349          self.retval = False
350  
351          root = Tk()
352  
353          def do_close():
354              root.destroy()
355  
356          def do_yes():
357              self.retval = True
358              do_close()
359  
360          phototmp = ImageTk.PhotoImage(Image.open(tmpfn))
361          photoref = ImageTk.PhotoImage(Image.open(reffn))
362          Label(root, text='The test %s\nhave generated an different'
363                'image as the reference one..' % testid).pack()
364          Label(root, text='Which one is good ?').pack()
365          Label(root, text=code, justify=LEFT).pack(side=RIGHT)
366          Label(root, image=phototmp).pack(side=RIGHT)
367          Label(root, image=photoref).pack(side=LEFT)
368          Button(root, text='Use the new image -->',
369                 command=do_yes).pack(side=BOTTOM)
370          Button(root, text='<-- Use the reference',
371                 command=do_close).pack(side=BOTTOM)
372          root.mainloop()
373  
374          return self.retval

377          '''Render the new frames and: 
378   
379          * tick the Clock 
380          * dispatch input from all registered providers 
381          * flush all the canvas operations 
382          * redraw Window canvas if necessary

384          from kivy.base import EventLoop

386              EventLoop.idle()

389  class UnitTestTouch(MotionEvent):
390      '''Custom MotionEvent representing a single touch. Similar to `on_touch_*` 
391      methods from the Widget class, this one introduces: 
392   
393      * touch_down 
394      * touch_move 
395      * touch_up 
396   
397      Create a new touch with:: 
398   
399          touch = UnitTestTouch(x, y) 
400   
401      then you press it on the default position with:: 
402   
403          touch.touch_down() 
404   
405      or move it or even release with these simple calls:: 
406   
407          touch.touch_move(new_x, new_y) 
408          touch.touch_up() 
409      '''
410  
411      def __init__(self, x, y):
412          '''Create a MotionEvent instance with X and Y of the first 
413          position a touch is at. 
414          '''
415  
416          from kivy.base import EventLoop
417          self.eventloop = EventLoop
418          win = EventLoop.window
419  
420          super(UnitTestTouch, self).__init__(
421              # device, (tuio) id, args
422              self.__class__.__name__, 99, {
423                  "x": x / float(win.width),
424                  "y": y / float(win.height),
425              }
426          )
427  
428      def touch_down(self, *args):
429          self.eventloop.post_dispatch_input("begin", self)
430  
431      def touch_move(self, x, y):
432          win = self.eventloop.window
433          self.move({
434              "x": x / float(win.width),
435              "y": y / float(win.height)
436          })
437          self.eventloop.post_dispatch_input("update", self)
438  
439      def touch_up(self, *args):
440          self.eventloop.post_dispatch_input("end", self)
441  
442      def depack(self, args):
443          # set MotionEvent to touch
444          self.is_touch = True
445  
446          # set sx/sy properties to ratio (e.g. X / win.width)
447          self.sx = args['x']
448          self.sy = args['y']
449  
450          # set profile to accept x, y and pos properties
451          self.profile = ['pos']
452  
453          # run depack after we set the values
454          super(UnitTestTouch, self).depack(args)

466  def async_run(func=None, app_cls_func=None):
467      def inner_func(func):
468          if 'mock' == cgl_get_backend_name():
469              return pytest.mark.skip(
470                  reason='Skipping because gl backend is set to mock')(func)
471  
472          if sys.version_info[0] < 3 or sys.version_info[1] <= 5:
473              return pytest.mark.skip(
474                  reason='Skipping because graphics tests are not supported on '
475                         'py3.5, only on py3.6+')(func)
476  
477          if app_cls_func is not None:
478              func = pytest.mark.parametrize(
479                  "kivy_app", [[app_cls_func], ], indirect=True)(func)
480  
481          if kivy_eventloop == 'asyncio':
482              try:
483                  import pytest_asyncio
484                  return pytest.mark.asyncio(func)
485              except ImportError:
486                  return pytest.mark.skip(
487                      reason='KIVY_EVENTLOOP == "asyncio" but '
488                             '"pytest-asyncio" is not installed')(func)
489          elif kivy_eventloop == 'trio':
490              try:
491                  import trio
492                  from pytest_trio import trio_fixture
493                  func._force_trio_fixture = True
494                  return func
495              except ImportError:
496                  return pytest.mark.skip(
497                      reason='KIVY_EVENTLOOP == "trio" but '
498                             '"pytest-trio" is not installed')(func)
499          else:
500              return pytest.mark.skip(
501                  reason='KIVY_EVENTLOOP must be set to either of "asyncio" or '
502                         '"trio" to run async tests')(func)
503  
504      if func is None:
505          return inner_func
506  
507      return inner_func(func)
