2    Color Picker 
3    ============ 
4     
5    .. versionadded:: 1.7.0 
6     
7    .. warning:: 
8     
9        This widget is experimental. Its use and API can change at any time until 
10       this warning is removed. 
11    
12   .. image:: images/colorpicker.png 
13       :align: right 
14    
15   The ColorPicker widget allows a user to select a color from a chromatic 
16   wheel where pinch and zoom can be used to change the wheel's saturation. 
17   Sliders and TextInputs are also provided for entering the RGBA/HSV/HEX values 
18   directly. 
19    
20   Usage:: 
21    
22       clr_picker = ColorPicker() 
23       parent.add_widget(clr_picker) 
24    
25       # To monitor changes, we can bind to color property changes 
26       def on_color(instance, value): 
27           print "RGBA = ", str(value)  #  or instance.color 
28           print "HSV = ", str(instance.hsv) 
29           print "HEX = ", str(instance.hex_color) 
30    
31       clr_picker.bind(color=on_color) 
32

52   def distance(pt1, pt2):
53       return sqrt((pt1[0] - pt2[0]) ** 2. + (pt1[1] - pt2[1]) ** 2.)
54   
55   
56   def polar_to_rect(origin, r, theta):
57       return origin[0] + r * cos(theta), origin[1] + r * sin(theta)
58   
59   
60   def rect_to_polar(origin, x, y):
61       if x == origin[0]:
62           if y == origin[1]:
63               return (0, 0)
64           elif y > origin[1]:
65               return (y - origin[1], pi / 2.)
66           else:
67               return (origin[1] - y, 3 * pi / 2.)
68       t = atan(float((y - origin[1])) / (x - origin[0]))
69       if x - origin[0] < 0:
70           t += pi
71   
72       if t < 0:
73           t += 2 * pi
74   
75       return (distance((x, y), origin), t)

78   class ColorWheel(Widget):
79       '''Chromatic wheel for the ColorPicker. 
80    
81       .. versionchanged:: 1.7.1 
82           `font_size`, `font_name` and `foreground_color` have been removed. The 
83           sizing is now the same as others widget, based on 'sp'. Orientation is 
84           also automatically determined according to the width/height ratio. 
85    
86       '''
87   
88       r = BoundedNumericProperty(0, min=0, max=1)
89       '''The Red value of the color currently selected. 
90    
91       :attr:`r` is a :class:`~kivy.properties.BoundedNumericProperty` and 
92       can be a value from 0 to 1. It defaults to 0. 
93       '''
94   
95       g = BoundedNumericProperty(0, min=0, max=1)
96       '''The Green value of the color currently selected. 
97    
98       :attr:`g` is a :class:`~kivy.properties.BoundedNumericProperty` 
99       and can be a value from 0 to 1. 
100      '''
101  
102      b = BoundedNumericProperty(0, min=0, max=1)
103      '''The Blue value of the color currently selected. 
104   
105      :attr:`b` is a :class:`~kivy.properties.BoundedNumericProperty` and 
106      can be a value from 0 to 1. 
107      '''
108  
109      a = BoundedNumericProperty(0, min=0, max=1)
110      '''The Alpha value of the color currently selected. 
111   
112      :attr:`a` is a :class:`~kivy.properties.BoundedNumericProperty` and 
113      can be a value from 0 to 1. 
114      '''
115  
116      color = ReferenceListProperty(r, g, b, a)
117      '''The holds the color currently selected. 
118   
119      :attr:`color` is a :class:`~kivy.properties.ReferenceListProperty` and 
120      contains a list of `r`, `g`, `b`, `a` values. 
121      '''
122  
123      _origin = ListProperty((100, 100))
124      _radius = NumericProperty(100)
125  
126      _piece_divisions = NumericProperty(10)
127      _pieces_of_pie = NumericProperty(16)
128  
129      _inertia_slowdown = 1.25
130      _inertia_cutoff = .25
131  
132      _num_touches = 0
133      _pinch_flag = False
134  
135      _hsv = ListProperty([1, 1, 1, 0])
136  
137      def __init__(self, **kwargs):
138          super(ColorWheel, self).__init__(**kwargs)
139  
140          pdv = self._piece_divisions
141          self.sv_s = [(float(x) / pdv, 1) for x in range(pdv)] + [
142              (1, float(y) / pdv) for y in reversed(range(pdv))]
143  
144      def on__origin(self, instance, value):
145          self.init_wheel(None)
146  
147      def on__radius(self, instance, value):
148          self.init_wheel(None)
149  
150      def init_wheel(self, dt):
151          # initialize list to hold all meshes
152          self.canvas.clear()
153          self.arcs = []
154          self.sv_idx = 0
155          pdv = self._piece_divisions
156          ppie = self._pieces_of_pie
157  
158          for r in range(pdv):
159              for t in range(ppie):
160                  self.arcs.append(
161                      _ColorArc(
162                          self._radius * (float(r) / float(pdv)),
163                          self._radius * (float(r + 1) / float(pdv)),
164                          2 * pi * (float(t) / float(ppie)),
165                          2 * pi * (float(t + 1) / float(ppie)),
166                          origin=self._origin,
167                          color=(float(t) / ppie,
168                                 self.sv_s[self.sv_idx + r][0],
169                                 self.sv_s[self.sv_idx + r][1],
170                                 1)))
171  
172                  self.canvas.add(self.arcs[-1])
173  
174      def recolor_wheel(self):
175          ppie = self._pieces_of_pie
176          for idx, segment in enumerate(self.arcs):
177              segment.change_color(
178                  sv=self.sv_s[int(self.sv_idx + idx / ppie)])
179  
180      def change_alpha(self, val):
181          for idx, segment in enumerate(self.arcs):
182              segment.change_color(a=val)
183  
184      def inertial_incr_sv_idx(self, dt):
185          # if its already zoomed all the way out, cancel the inertial zoom
186          if self.sv_idx == len(self.sv_s) - self._piece_divisions:
187              return False
188  
189          self.sv_idx += 1
190          self.recolor_wheel()
191          if dt * self._inertia_slowdown > self._inertia_cutoff:
192              return False
193          else:
194              Clock.schedule_once(self.inertial_incr_sv_idx,
195                                  dt * self._inertia_slowdown)
196  
197      def inertial_decr_sv_idx(self, dt):
198          # if its already zoomed all the way in, cancel the inertial zoom
199          if self.sv_idx == 0:
200              return False
201          self.sv_idx -= 1
202          self.recolor_wheel()
203          if dt * self._inertia_slowdown > self._inertia_cutoff:
204              return False
205          else:
206              Clock.schedule_once(self.inertial_decr_sv_idx,
207                                  dt * self._inertia_slowdown)
208  
209      def on_touch_down(self, touch):
210          r = self._get_touch_r(touch.pos)
211          if r > self._radius:
212              return False
213  
214          # code is still set up to allow pinch to zoom, but this is
215          # disabled for now since it was fiddly with small wheels.
216          # Comment out these lines and  adjust on_touch_move to reenable
217          # this.
218          if self._num_touches != 0:
219              return False
220  
221          touch.grab(self)
222          self._num_touches += 1
223          touch.ud['anchor_r'] = r
224          touch.ud['orig_sv_idx'] = self.sv_idx
225          touch.ud['orig_time'] = Clock.get_time()
226  
227      def on_touch_move(self, touch):
228          if touch.grab_current is not self:
229              return
230          r = self._get_touch_r(touch.pos)
231          goal_sv_idx = (touch.ud['orig_sv_idx'] -
232                         int((r - touch.ud['anchor_r']) /
233                              (float(self._radius) / self._piece_divisions)))
234  
235          if (
236              goal_sv_idx != self.sv_idx and
237              goal_sv_idx >= 0 and
238              goal_sv_idx <= len(self.sv_s) - self._piece_divisions
239          ):
240              # this is a pinch to zoom
241              self._pinch_flag = True
242              self.sv_idx = goal_sv_idx
243              self.recolor_wheel()
244  
245      def on_touch_up(self, touch):
246          if touch.grab_current is not self:
247              return
248          touch.ungrab(self)
249          self._num_touches -= 1
250          if self._pinch_flag:
251              if self._num_touches == 0:
252                  # user was pinching, and now both fingers are up. Return
253                  # to normal
254                  if self.sv_idx > touch.ud['orig_sv_idx']:
255                      Clock.schedule_once(
256                          self.inertial_incr_sv_idx,
257                          (Clock.get_time() - touch.ud['orig_time']) /
258                          (self.sv_idx - touch.ud['orig_sv_idx']))
259  
260                  if self.sv_idx < touch.ud['orig_sv_idx']:
261                      Clock.schedule_once(
262                          self.inertial_decr_sv_idx,
263                          (Clock.get_time() - touch.ud['orig_time']) /
264                          (self.sv_idx - touch.ud['orig_sv_idx']))
265  
266                  self._pinch_flag = False
267                  return
268              else:
269                  # user was pinching, and at least one finger remains. We
270                  # don't want to treat the remaining fingers as touches
271                  return
272          else:
273              r, theta = rect_to_polar(self._origin, *touch.pos)
274              # if touch up is outside the wheel, ignore
275              if r >= self._radius:
276                  return
277              # compute which ColorArc is being touched (they aren't
278              # widgets so we don't get collide_point) and set
279              # _hsv based on the selected ColorArc
280              piece = int((theta / (2 * pi)) * self._pieces_of_pie)
281              division = int((r / self._radius) * self._piece_divisions)
282              hsva = list(
283                  self.arcs[self._pieces_of_pie * division + piece].color)
284              self.color = list(hsv_to_rgb(*hsva[:3])) + hsva[-1:]
285  
286      def _get_touch_r(self, pos):
287          return distance(pos, self._origin)

290  class _ColorArc(InstructionGroup):
291      def __init__(self, r_min, r_max, theta_min, theta_max,
292                   color=(0, 0, 1, 1), origin=(0, 0), **kwargs):
293          super(_ColorArc, self).__init__(**kwargs)
294          self.origin = origin
295          self.r_min = r_min
296          self.r_max = r_max
297          self.theta_min = theta_min
298          self.theta_max = theta_max
299          self.color = color
300          self.color_instr = Color(*color, mode='hsv')
301          self.add(self.color_instr)
302          self.mesh = self.get_mesh()
303          self.add(self.mesh)
304  
305      def __str__(self):
306          return "r_min: %s r_max: %s theta_min: %s theta_max: %s color: %s" % (
307              self.r_min, self.r_max, self.theta_min, self.theta_max, self.color
308          )
309  
310      def get_mesh(self):
311          v = []
312          # first calculate the distance between endpoints of the outer
313          # arc, so we know how many steps to use when calculating
314          # vertices
315          theta_step_outer = 0.1
316          theta = self.theta_max - self.theta_min
317          d_outer = int(theta / theta_step_outer)
318          theta_step_outer = theta / d_outer
319  
320          if self.r_min == 0:
321              for x in range(0, d_outer, 2):
322                  v += (polar_to_rect(self.origin, self.r_max,
323                                      self.theta_min + x * theta_step_outer
324                                      ) * 2)
325                  v += polar_to_rect(self.origin, 0, 0) * 2
326                  v += (polar_to_rect(self.origin, self.r_max,
327                                      self.theta_min + (x + 1) * theta_step_outer
328                                      ) * 2)
329              if not d_outer & 1:  # add a last point if d_outer is even
330                  v += (polar_to_rect(self.origin, self.r_max,
331                                      self.theta_min + d_outer * theta_step_outer
332                                      ) * 2)
333          else:
334              for x in range(d_outer + 1):
335                  v += (polar_to_rect(self.origin, self.r_min,
336                                      self.theta_min + x * theta_step_outer
337                                      ) * 2)
338                  v += (polar_to_rect(self.origin, self.r_max,
339                                      self.theta_min + x * theta_step_outer
340                                      ) * 2)
341  
342          return Mesh(vertices=v, indices=range(int(len(v) / 4)),
343                      mode='triangle_strip')
344  
345      def change_color(self, color=None, color_delta=None, sv=None, a=None):
346          self.remove(self.color_instr)
347          if color is not None:
348              self.color = color
349          elif color_delta is not None:
350              self.color = [self.color[i] + color_delta[i] for i in range(4)]
351          elif sv is not None:
352              self.color = (self.color[0], sv[0], sv[1], self.color[3])
353          elif a is not None:
354              self.color = (self.color[0], self.color[1], self.color[2], a)
355          self.color_instr = Color(*self.color, mode='hsv')
356          self.insert(0, self.color_instr)

359  class ColorPicker(RelativeLayout):
360      ''' 
361      See module documentation. 
362      '''
363  
364      font_name = StringProperty('data/fonts/RobotoMono-Regular.ttf')
365      '''Specifies the font used on the ColorPicker. 
366   
367      :attr:`font_name` is a :class:`~kivy.properties.StringProperty` and 
368      defaults to 'data/fonts/RobotoMono-Regular.ttf'. 
369      '''
370  
371      color = ListProperty((1, 1, 1, 1))
372      '''The :attr:`color` holds the color currently selected in rgba format. 
373   
374      :attr:`color` is a :class:`~kivy.properties.ListProperty` and defaults to 
375      (1, 1, 1, 1). 
376      '''
377  
378      def _get_hsv(self):
379          return rgb_to_hsv(*self.color[:3])
380  
381      def _set_hsv(self, value):
382          if self._updating_clr:
383              return
384          self.set_color(value)
385  
386      hsv = AliasProperty(_get_hsv, _set_hsv, bind=('color', ))
387      '''The :attr:`hsv` holds the color currently selected in hsv format. 
388   
389      :attr:`hsv` is a :class:`~kivy.properties.ListProperty` and defaults to 
390      (1, 1, 1). 
391      '''
392      def _get_hex(self):
393          return get_hex_from_color(self.color)
394  
395      def _set_hex(self, value):
396          if self._updating_clr:
397              return
398          self.set_color(get_color_from_hex(value)[:4])
399  
400      hex_color = AliasProperty(_get_hex, _set_hex, bind=('color',), cache=True)
401      '''The :attr:`hex_color` holds the currently selected color in hex. 
402   
403      :attr:`hex_color` is an :class:`~kivy.properties.AliasProperty` and 
404      defaults to `#ffffffff`. 
405      '''
406  
407      wheel = ObjectProperty(None)
408      '''The :attr:`wheel` holds the color wheel. 
409   
410      :attr:`wheel` is an :class:`~kivy.properties.ObjectProperty` and 
411      defaults to None. 
412      '''
413  
414      _update_clr_ev = _update_hex_ev = None
415  
416      # now used only internally.
417      foreground_color = ListProperty((1, 1, 1, 1))
418  
419      def _trigger_update_clr(self, mode, clr_idx, text):
420          if self._updating_clr:
421              return
422          self._updating_clr = True
423          self._upd_clr_list = mode, clr_idx, text
424          ev = self._update_clr_ev
425          if ev is None:
426              ev = self._update_clr_ev = Clock.create_trigger(self._update_clr)
427          ev()
428  
429      def _update_clr(self, dt):
430          # to prevent interaction between hsv/rgba, we work internaly using rgba
431          mode, clr_idx, text = self._upd_clr_list
432          try:
433              text = min(255, max(0, float(text)))
434              if mode == 'rgb':
435                  self.color[clr_idx] = float(text) / 255.
436              else:
437                  hsv = list(self.hsv[:])
438                  hsv[clr_idx] = float(text) / 255.
439                  self.color[:3] = hsv_to_rgb(*hsv)
440          except ValueError:
441              Logger.warning('ColorPicker: invalid value : {}'.format(text))
442          finally:
443              self._updating_clr = False
444  
445      def _update_hex(self, dt):
446          try:
447              if len(self._upd_hex_list) != 9:
448                  return
449              self._updating_clr = False
450              self.hex_color = self._upd_hex_list
451          finally:
452              self._updating_clr = False
453  
454      def _trigger_update_hex(self, text):
455          if self._updating_clr:
456              return
457          self._updating_clr = True
458          self._upd_hex_list = text
459          ev = self._update_hex_ev
460          if ev is None:
461              ev = self._update_hex_ev = Clock.create_trigger(self._update_hex)
462          ev()
463  
464      def set_color(self, color):
465          self._updating_clr = True
466          if len(color) == 3:
467              self.color[:3] = color
468          else:
469              self.color = color
470          self._updating_clr = False
471  
472      def __init__(self, **kwargs):
473          self._updating_clr = False
474          super(ColorPicker, self).__init__(**kwargs)
