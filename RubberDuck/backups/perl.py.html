207      def analyse_text(text):
208          if shebang_matches(text, r'perl'):
209              return True
210  
211          result = 0
212  
213          if re.search(r'(?:my|our)\s+[$@%(]', text):
214              result += 0.9
215  
216          if ':=' in text:
217              # := is not valid Perl, but it appears in unicon, so we should
218              # become less confident if we think we found Perl with :=
219              result /= 2
220  
221          return result

493      def _build_word_match(words, boundary_regex_fragment=None, prefix='', suffix=''):
494          if boundary_regex_fragment is None:
495              return r'\b(' + prefix + r'|'.join(re.escape(x) for x in words) + \
496                  suffix + r')\b'
497          else:
498              return r'(?<!' + boundary_regex_fragment + r')' + prefix + r'(' + \
499                  r'|'.join(re.escape(x) for x in words) + r')' + suffix + r'(?!' + \
500                  boundary_regex_fragment + r')'

502      def brackets_callback(token_class):
503          def callback(lexer, match, context):
504              groups = match.groupdict()
505              opening_chars = groups['delimiter']
506              n_chars = len(opening_chars)
507              adverbs = groups.get('adverbs')
508  
509              closer = Perl6Lexer.PERL6_BRACKETS.get(opening_chars[0])
510              text = context.text
511  
512              if closer is None:  # it's not a mirrored character, which means we
513                                  # just need to look for the next occurrence
514  
515                  end_pos = text.find(opening_chars, match.start('delimiter') + n_chars)
516              else:   # we need to look for the corresponding closing character,
517                      # keep nesting in mind
518                  closing_chars = closer * n_chars
519                  nesting_level = 1
520  
521                  search_pos = match.start('delimiter')
522  
523                  while nesting_level > 0:
524                      next_open_pos = text.find(opening_chars, search_pos + n_chars)
525                      next_close_pos = text.find(closing_chars, search_pos + n_chars)
526  
527                      if next_close_pos == -1:
528                          next_close_pos = len(text)
529                          nesting_level = 0
530                      elif next_open_pos != -1 and next_open_pos < next_close_pos:
531                          nesting_level += 1
532                          search_pos = next_open_pos
533                      else:  # next_close_pos < next_open_pos
534                          nesting_level -= 1
535                          search_pos = next_close_pos
536  
537                  end_pos = next_close_pos
538  
539              if end_pos < 0:     # if we didn't find a closer, just highlight the
540                                  # rest of the text in this class
541                  end_pos = len(text)
542  
543              if adverbs is not None and re.search(r':to\b', adverbs):
544                  heredoc_terminator = text[match.start('delimiter') + n_chars:end_pos]
545                  end_heredoc = re.search(r'^\s*' + re.escape(heredoc_terminator) +
546                                          r'\s*$', text[end_pos:], re.MULTILINE)
547  
548                  if end_heredoc:
549                      end_pos += end_heredoc.end()
550                  else:
551                      end_pos = len(text)
552  
553              yield match.start(), token_class, text[match.start():end_pos + n_chars]
554              context.pos = end_pos + n_chars
555  
556          return callback

558      def opening_brace_callback(lexer, match, context):
559          stack = context.stack
560  
561          yield match.start(), Text, context.text[match.start():match.end()]
562          context.pos = match.end()
563  
564          # if we encounter an opening brace and we're one level
565          # below a token state, it means we need to increment
566          # the nesting level for braces so we know later when
567          # we should return to the token rules.
568          if len(stack) > 2 and stack[-2] == 'token':
569              context.perl6_token_nesting_level += 1
570  
571      def closing_brace_callback(lexer, match, context):
572          stack = context.stack
573  
574          yield match.start(), Text, context.text[match.start():match.end()]
575          context.pos = match.end()
576  
577          # if we encounter a free closing brace and we're one level
578          # below a token state, it means we need to check the nesting
579          # level to see if we need to return to the token state.
580          if len(stack) > 2 and stack[-2] == 'token':
581              context.perl6_token_nesting_level -= 1
582              if context.perl6_token_nesting_level == 0:
583                  stack.pop()

585      def embedded_perl6_callback(lexer, match, context):
586          context.perl6_token_nesting_level = 1
587          yield match.start(), Text, context.text[match.start():match.end()]
588          context.pos = match.end()
589          context.stack.append('root')

674      def analyse_text(text):
675          def strip_pod(lines):
676              in_pod = False
677              stripped_lines = []
678  
679              for line in lines:
680                  if re.match(r'^=(?:end|cut)', line):
681                      in_pod = False
682                  elif re.match(r'^=\w+', line):
683                      in_pod = True
684                  elif not in_pod:
685                      stripped_lines.append(line)
686  
687              return stripped_lines
688  
689          # XXX handle block comments
690          lines = text.splitlines()
691          lines = strip_pod(lines)
692          text = '\n'.join(lines)
693  
694          if shebang_matches(text, r'perl6|rakudo|niecza|pugs'):
695              return True
696  
697          saw_perl_decl = False
698          rating = False
699  
700          # check for my/our/has declarations
701          if re.search(r"(?:my|our|has)\s+(?:" + Perl6Lexer.PERL6_IDENTIFIER_RANGE +
702                       r"+\s+)?[$@%&(]", text):
703              rating = 0.8
704              saw_perl_decl = True
705  
706          for line in lines:
707              line = re.sub('#.*', '', line)
708              if re.match(r'^\s*$', line):
709                  continue
710  
711              # match v6; use v6; use v6.0; use v6.0.0;
712              if re.match(r'^\s*(?:use\s+)?v6(?:\.\d(?:\.\d)?)?;', line):
713                  return True
714              # match class, module, role, enum, grammar declarations
715              class_decl = re.match(r'^\s*(?:(?P<scope>my|our)\s+)?(?:module|class|role|enum|grammar)', line)
716              if class_decl:
717                  if saw_perl_decl or class_decl.group('scope') is not None:
718                      return True
719                  rating = 0.05
720                  continue
721              break
722  
723          if ':=' in text:
724              # Same logic as above for PerlLexer
725              rating /= 2
726  
727          return rating
