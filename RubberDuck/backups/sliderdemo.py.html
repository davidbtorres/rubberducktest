4    import win32con, win32ui
5    from pywin.mfc import dialog
6    
7    
8    class MyDialog(dialog.Dialog):
9        """ 
10       Example using simple controls 
11       """
12   
13       _dialogstyle = (
14           win32con.WS_MINIMIZEBOX
15           | win32con.WS_DLGFRAME
16           | win32con.DS_MODALFRAME
17           | win32con.WS_POPUP
18           | win32con.WS_VISIBLE
19           | win32con.WS_CAPTION
20           | win32con.WS_SYSMENU
21           | win32con.DS_SETFONT
22       )
23       _buttonstyle = (
24           win32con.BS_PUSHBUTTON
25           | win32con.WS_TABSTOP
26           | win32con.WS_CHILD
27           | win32con.WS_VISIBLE
28       )
29       ### The static template, contains all "normal" dialog items
30       DIALOGTEMPLATE = [
31           # the dialog itself is the first element in the template
32           ["Example slider", (0, 0, 50, 43), _dialogstyle, None, (8, "MS SansSerif")],
33           # rest of elements are the controls within the dialog
34           # standard "Close" button
35           [128, "Close", win32con.IDCANCEL, (0, 30, 50, 13), _buttonstyle],
36       ]
37       ### ID of the control to be created during dialog initialisation
38       IDC_SLIDER = 9500
39   
40       def __init__(self):
41           dialog.Dialog.__init__(self, self.DIALOGTEMPLATE)
42   
43       def OnInitDialog(self):
44           rc = dialog.Dialog.OnInitDialog(self)
45           # now initialise your controls that you want to create
46           # programmatically, including those which are OLE controls
47           # those created directly by win32ui.Create*
48           # and your "custom controls" which are subclasses/whatever
49           win32ui.EnableControlContainer()
50           self.slider = win32ui.CreateSliderCtrl()
51           self.slider.CreateWindow(
52               win32con.WS_TABSTOP | win32con.WS_VISIBLE,
53               (0, 0, 100, 30),
54               self._obj_,
55               self.IDC_SLIDER,
56           )
57           self.HookMessage(self.OnSliderMove, win32con.WM_HSCROLL)
58           return rc
59   
60       def OnSliderMove(self, params):
61           print("Slider moved")
62   
63       def OnCancel(self):
64           print("The slider control is at position", self.slider.GetPos())
65           self._obj_.OnCancel()
66   
67   
68   ###
69   def demo():
70       dia = MyDialog()
71       dia.DoModal()
72   
73   
74   if __name__ == "__main__":
75       demo()
