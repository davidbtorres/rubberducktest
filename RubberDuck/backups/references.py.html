18   class PropagateTargets(Transform):
19   
20       """ 
21       Propagate empty internal targets to the next element. 
22    
23       Given the following nodes:: 
24    
25           <target ids="internal1" names="internal1"> 
26           <target anonymous="1" ids="id1"> 
27           <target ids="internal2" names="internal2"> 
28           <paragraph> 
29               This is a test. 
30    
31       PropagateTargets propagates the ids and names of the internal 
32       targets preceding the paragraph to the paragraph itself:: 
33    
34           <target refid="internal1"> 
35           <target anonymous="1" refid="id1"> 
36           <target refid="internal2"> 
37           <paragraph ids="internal2 id1 internal1" names="internal2 internal1"> 
38               This is a test. 
39       """
40   
41       default_priority = 260
42   
43       def apply(self):
44           for target in self.document.traverse(nodes.target):
45               # Only block-level targets without reference (like ".. _target:"):
46               if (isinstance(target.parent, nodes.TextElement) or
47                   (target.hasattr('refid') or target.hasattr('refuri') or
48                    target.hasattr('refname'))):
49                   continue
50               assert len(target) == 0, 'error: block-level target has children'
51               next_node = target.next_node(ascend=True)
52               # skip system messages (may be removed by universal.FilterMessages)
53               while isinstance(next_node, nodes.system_message):
54                   next_node = next_node.next_node(ascend=True, descend=False)
55               # Do not move names and ids into Invisibles (we'd lose the
56               # attributes) or different Targetables (e.g. footnotes).
57               if (next_node is None
58                   or isinstance(next_node, (nodes.Invisible, nodes.Targetable))
59                   and not isinstance(next_node, nodes.target)):
60                   continue
61               next_node['ids'].extend(target['ids'])
62               next_node['names'].extend(target['names'])
63               # Set defaults for next_node.expect_referenced_by_name/id.
64               if not hasattr(next_node, 'expect_referenced_by_name'):
65                   next_node.expect_referenced_by_name = {}
66               if not hasattr(next_node, 'expect_referenced_by_id'):
67                   next_node.expect_referenced_by_id = {}
68               for id in target['ids']:
69                   # Update IDs to node mapping.
70                   self.document.ids[id] = next_node
71                   # If next_node is referenced by id ``id``, this
72                   # target shall be marked as referenced.
73                   next_node.expect_referenced_by_id[id] = target
74               for name in target['names']:
75                   next_node.expect_referenced_by_name[name] = target
76               # If there are any expect_referenced_by_... attributes
77               # in target set, copy them to next_node.
78               next_node.expect_referenced_by_name.update(
79                   getattr(target, 'expect_referenced_by_name', {}))
80               next_node.expect_referenced_by_id.update(
81                   getattr(target, 'expect_referenced_by_id', {}))
82               # Set refid to point to the first former ID of target
83               # which is now an ID of next_node.
84               target['refid'] = target['ids'][0]
85               # Clear ids and names; they have been moved to
86               # next_node.
87               target['ids'] = []
88               target['names'] = []
89               self.document.note_refid(target)

92   class AnonymousHyperlinks(Transform):
93   
94       """ 
95       Link anonymous references to targets.  Given:: 
96    
97           <paragraph> 
98               <reference anonymous="1"> 
99                   internal 
100              <reference anonymous="1"> 
101                  external 
102          <target anonymous="1" ids="id1"> 
103          <target anonymous="1" ids="id2" refuri="http://external"> 
104   
105      Corresponding references are linked via "refid" or resolved via "refuri":: 
106   
107          <paragraph> 
108              <reference anonymous="1" refid="id1"> 
109                  text 
110              <reference anonymous="1" refuri="http://external"> 
111                  external 
112          <target anonymous="1" ids="id1"> 
113          <target anonymous="1" ids="id2" refuri="http://external"> 
114      """
115  
116      default_priority = 440
117  
118      def apply(self):
119          anonymous_refs = []
120          anonymous_targets = []
121          for node in self.document.traverse(nodes.reference):
122              if node.get('anonymous'):
123                  anonymous_refs.append(node)
124          for node in self.document.traverse(nodes.target):
125              if node.get('anonymous'):
126                  anonymous_targets.append(node)
127          if len(anonymous_refs) \
128                != len(anonymous_targets):
129              msg = self.document.reporter.error(
130                    'Anonymous hyperlink mismatch: %s references but %s '
131                    'targets.\nSee "backrefs" attribute for IDs.'
132                    % (len(anonymous_refs), len(anonymous_targets)))
133              msgid = self.document.set_id(msg)
134              for ref in anonymous_refs:
135                  prb = nodes.problematic(
136                        ref.rawsource, ref.rawsource, refid=msgid)
137                  prbid = self.document.set_id(prb)
138                  msg.add_backref(prbid)
139                  ref.replace_self(prb)
140              return
141          for ref, target in zip(anonymous_refs, anonymous_targets):
142              target.referenced = 1
143              while True:
144                  if target.hasattr('refuri'):
145                      ref['refuri'] = target['refuri']
146                      ref.resolved = 1
147                      break
148                  else:
149                      if not target['ids']:
150                          # Propagated target.
151                          target = self.document.ids[target['refid']]
152                          continue
153                      ref['refid'] = target['ids'][0]
154                      self.document.note_refid(ref)
155                      break

158  class IndirectHyperlinks(Transform):
159  
160      """ 
161      a) Indirect external references:: 
162   
163             <paragraph> 
164                 <reference refname="indirect external"> 
165                     indirect external 
166             <target id="id1" name="direct external" 
167                 refuri="http://indirect"> 
168             <target id="id2" name="indirect external" 
169                 refname="direct external"> 
170   
171         The "refuri" attribute is migrated back to all indirect targets 
172         from the final direct target (i.e. a target not referring to 
173         another indirect target):: 
174   
175             <paragraph> 
176                 <reference refname="indirect external"> 
177                     indirect external 
178             <target id="id1" name="direct external" 
179                 refuri="http://indirect"> 
180             <target id="id2" name="indirect external" 
181                 refuri="http://indirect"> 
182   
183         Once the attribute is migrated, the preexisting "refname" attribute 
184         is dropped. 
185   
186      b) Indirect internal references:: 
187   
188             <target id="id1" name="final target"> 
189             <paragraph> 
190                 <reference refname="indirect internal"> 
191                     indirect internal 
192             <target id="id2" name="indirect internal 2" 
193                 refname="final target"> 
194             <target id="id3" name="indirect internal" 
195                 refname="indirect internal 2"> 
196   
197         Targets which indirectly refer to an internal target become one-hop 
198         indirect (their "refid" attributes are directly set to the internal 
199         target's "id"). References which indirectly refer to an internal 
200         target become direct internal references:: 
201   
202             <target id="id1" name="final target"> 
203             <paragraph> 
204                 <reference refid="id1"> 
205                     indirect internal 
206             <target id="id2" name="indirect internal 2" refid="id1"> 
207             <target id="id3" name="indirect internal" refid="id1"> 
208      """
209  
210      default_priority = 460
211  
212      def apply(self):
213          for target in self.document.indirect_targets:
214              if not target.resolved:
215                  self.resolve_indirect_target(target)
216              self.resolve_indirect_references(target)
217  
218      def resolve_indirect_target(self, target):
219          refname = target.get('refname')
220          if refname is None:
221              reftarget_id = target['refid']
222          else:
223              reftarget_id = self.document.nameids.get(refname)
224              if not reftarget_id:
225                  # Check the unknown_reference_resolvers
226                  for resolver_function in \
227                          self.document.transformer.unknown_reference_resolvers:
228                      if resolver_function(target):
229                          break
230                  else:
231                      self.nonexistent_indirect_target(target)
232                  return
233          reftarget = self.document.ids[reftarget_id]
234          reftarget.note_referenced_by(id=reftarget_id)
235          if isinstance(reftarget, nodes.target) \
236                 and not reftarget.resolved and reftarget.hasattr('refname'):
237              if hasattr(target, 'multiply_indirect'):
238                  #and target.multiply_indirect):
239                  #del target.multiply_indirect
240                  self.circular_indirect_reference(target)
241                  return
242              target.multiply_indirect = 1
243              self.resolve_indirect_target(reftarget) # multiply indirect
244              del target.multiply_indirect
245          if reftarget.hasattr('refuri'):
246              target['refuri'] = reftarget['refuri']
247              if 'refid' in target:
248                  del target['refid']
249          elif reftarget.hasattr('refid'):
250              target['refid'] = reftarget['refid']
251              self.document.note_refid(target)
252          else:
253              if reftarget['ids']:
254                  target['refid'] = reftarget_id
255                  self.document.note_refid(target)
256              else:
257                  self.nonexistent_indirect_target(target)
258                  return
259          if refname is not None:
260              del target['refname']
261          target.resolved = 1
262  
263      def nonexistent_indirect_target(self, target):
264          if target['refname'] in self.document.nameids:
265              self.indirect_target_error(target, 'which is a duplicate, and '
266                                         'cannot be used as a unique reference')
267          else:
268              self.indirect_target_error(target, 'which does not exist')
269  
270      def circular_indirect_reference(self, target):
271          self.indirect_target_error(target, 'forming a circular reference')
272  
273      def indirect_target_error(self, target, explanation):
274          naming = ''
275          reflist = []
276          if target['names']:
277              naming = '"%s" ' % target['names'][0]
278          for name in target['names']:
279              reflist.extend(self.document.refnames.get(name, []))
280          for id in target['ids']:
281              reflist.extend(self.document.refids.get(id, []))
282          if target['ids']:
283              naming += '(id="%s")' % target['ids'][0]
284          msg = self.document.reporter.error(
285                'Indirect hyperlink target %s refers to target "%s", %s.'
286                % (naming, target['refname'], explanation), base_node=target)
287          msgid = self.document.set_id(msg)
288          for ref in utils.uniq(reflist):
289              prb = nodes.problematic(
290                    ref.rawsource, ref.rawsource, refid=msgid)
291              prbid = self.document.set_id(prb)
292              msg.add_backref(prbid)
293              ref.replace_self(prb)
294          target.resolved = 1
295  
296      def resolve_indirect_references(self, target):
297          if target.hasattr('refid'):
298              attname = 'refid'
299              call_method = self.document.note_refid
300          elif target.hasattr('refuri'):
301              attname = 'refuri'
302              call_method = None
303          else:
304              return
305          attval = target[attname]
306          for name in target['names']:
307              reflist = self.document.refnames.get(name, [])
308              if reflist:
309                  target.note_referenced_by(name=name)
310              for ref in reflist:
311                  if ref.resolved:
312                      continue
313                  del ref['refname']
314                  ref[attname] = attval
315                  if call_method:
316                      call_method(ref)
317                  ref.resolved = 1
318                  if isinstance(ref, nodes.target):
319                      self.resolve_indirect_references(ref)
320          for id in target['ids']:
321              reflist = self.document.refids.get(id, [])
322              if reflist:
323                  target.note_referenced_by(id=id)
324              for ref in reflist:
325                  if ref.resolved:
326                      continue
327                  del ref['refid']
328                  ref[attname] = attval
329                  if call_method:
330                      call_method(ref)
331                  ref.resolved = 1
332                  if isinstance(ref, nodes.target):
333                      self.resolve_indirect_references(ref)

336  class ExternalTargets(Transform):
337  
338      """ 
339      Given:: 
340   
341          <paragraph> 
342              <reference refname="direct external"> 
343                  direct external 
344          <target id="id1" name="direct external" refuri="http://direct"> 
345   
346      The "refname" attribute is replaced by the direct "refuri" attribute:: 
347   
348          <paragraph> 
349              <reference refuri="http://direct"> 
350                  direct external 
351          <target id="id1" name="direct external" refuri="http://direct"> 
352      """
353  
354      default_priority = 640
355  
356      def apply(self):
357          for target in self.document.traverse(nodes.target):
358              if target.hasattr('refuri'):
359                  refuri = target['refuri']
360                  for name in target['names']:
361                      reflist = self.document.refnames.get(name, [])
362                      if reflist:
363                          target.note_referenced_by(name=name)
364                      for ref in reflist:
365                          if ref.resolved:
366                              continue
367                          del ref['refname']
368                          ref['refuri'] = refuri
369                          ref.resolved = 1

372  class InternalTargets(Transform):
373  
374      default_priority = 660
375  
376      def apply(self):
377          for target in self.document.traverse(nodes.target):
378              if not target.hasattr('refuri') and not target.hasattr('refid'):
379                  self.resolve_reference_ids(target)
380  
381      def resolve_reference_ids(self, target):
382          """ 
383          Given:: 
384   
385              <paragraph> 
386                  <reference refname="direct internal"> 
387                      direct internal 
388              <target id="id1" name="direct internal"> 
389   
390          The "refname" attribute is replaced by "refid" linking to the target's 
391          "id":: 
392   
393              <paragraph> 
394                  <reference refid="id1"> 
395                      direct internal 
396              <target id="id1" name="direct internal"> 
397          """
398          for name in target['names']:
399              refid = self.document.nameids.get(name)
400              reflist = self.document.refnames.get(name, [])
401              if reflist:
402                  target.note_referenced_by(name=name)
403              for ref in reflist:
404                  if ref.resolved:
405                      continue
406                  if refid:
407                      del ref['refname']
408                      ref['refid'] = refid
409                  ref.resolved = 1

412  class Footnotes(Transform):
413  
414      """ 
415      Assign numbers to autonumbered footnotes, and resolve links to footnotes, 
416      citations, and their references. 
417   
418      Given the following ``document`` as input:: 
419   
420          <document> 
421              <paragraph> 
422                  A labeled autonumbered footnote reference: 
423                  <footnote_reference auto="1" id="id1" refname="footnote"> 
424              <paragraph> 
425                  An unlabeled autonumbered footnote reference: 
426                  <footnote_reference auto="1" id="id2"> 
427              <footnote auto="1" id="id3"> 
428                  <paragraph> 
429                      Unlabeled autonumbered footnote. 
430              <footnote auto="1" id="footnote" name="footnote"> 
431                  <paragraph> 
432                      Labeled autonumbered footnote. 
433   
434      Auto-numbered footnotes have attribute ``auto="1"`` and no label. 
435      Auto-numbered footnote_references have no reference text (they're 
436      empty elements). When resolving the numbering, a ``label`` element 
437      is added to the beginning of the ``footnote``, and reference text 
438      to the ``footnote_reference``. 
439   
440      The transformed result will be:: 
441   
442          <document> 
443              <paragraph> 
444                  A labeled autonumbered footnote reference: 
445                  <footnote_reference auto="1" id="id1" refid="footnote"> 
446                      2 
447              <paragraph> 
448                  An unlabeled autonumbered footnote reference: 
449                  <footnote_reference auto="1" id="id2" refid="id3"> 
450                      1 
451              <footnote auto="1" id="id3" backrefs="id2"> 
452                  <label> 
453                      1 
454                  <paragraph> 
455                      Unlabeled autonumbered footnote. 
456              <footnote auto="1" id="footnote" name="footnote" backrefs="id1"> 
457                  <label> 
458                      2 
459                  <paragraph> 
460                      Labeled autonumbered footnote. 
461   
462      Note that the footnotes are not in the same order as the references. 
463   
464      The labels and reference text are added to the auto-numbered ``footnote`` 
465      and ``footnote_reference`` elements.  Footnote elements are backlinked to 
466      their references via "refids" attributes.  References are assigned "id" 
467      and "refid" attributes. 
468   
469      After adding labels and reference text, the "auto" attributes can be 
470      ignored. 
471      """
472  
473      default_priority = 620
474  
475      autofootnote_labels = None
476      """Keep track of unlabeled autonumbered footnotes."""
477  
478      symbols = [
479            # Entries 1-4 and 6 below are from section 12.51 of
480            # The Chicago Manual of Style, 14th edition.
481            '*',                          # asterisk/star
482            u'\u2020',                    #  &dagger; dagger
483            u'\u2021',                    #  &Dagger; double dagger
484            u'\u00A7',                    #  &sect; section mark
485            u'\u00B6',                    #  &para; paragraph mark (pilcrow)
486                                          # (parallels ['||'] in CMoS)
487            '#',                          # number sign
488            # The entries below were chosen arbitrarily.
489            u'\u2660',                    #  &spades; spade suit
490            u'\u2665',                    #  &hearts; heart suit
491            u'\u2666',                    #  &diams; diamond suit
492            u'\u2663',                    #  &clubs; club suit
493            ]
494  
495      def apply(self):
496          self.autofootnote_labels = []
497          startnum = self.document.autofootnote_start
498          self.document.autofootnote_start = self.number_footnotes(startnum)
499          self.number_footnote_references(startnum)
500          self.symbolize_footnotes()
501          self.resolve_footnotes_and_citations()
502  
503      def number_footnotes(self, startnum):
504          """ 
505          Assign numbers to autonumbered footnotes. 
506   
507          For labeled autonumbered footnotes, copy the number over to 
508          corresponding footnote references. 
509          """
510          for footnote in self.document.autofootnotes:
511              while True:
512                  label = str(startnum)
513                  startnum += 1
514                  if label not in self.document.nameids:
515                      break
516              footnote.insert(0, nodes.label('', label))
517              for name in footnote['names']:
518                  for ref in self.document.footnote_refs.get(name, []):
519                      ref += nodes.Text(label)
520                      ref.delattr('refname')
521                      assert len(footnote['ids']) == len(ref['ids']) == 1
522                      ref['refid'] = footnote['ids'][0]
523                      footnote.add_backref(ref['ids'][0])
524                      self.document.note_refid(ref)
525                      ref.resolved = 1
526              if not footnote['names'] and not footnote['dupnames']:
527                  footnote['names'].append(label)
528                  self.document.note_explicit_target(footnote, footnote)
529                  self.autofootnote_labels.append(label)
530          return startnum
531  
532      def number_footnote_references(self, startnum):
533          """Assign numbers to autonumbered footnote references."""
534          i = 0
535          for ref in self.document.autofootnote_refs:
536              if ref.resolved or ref.hasattr('refid'):
537                  continue
538              try:
539                  label = self.autofootnote_labels[i]
540              except IndexError:
541                  msg = self.document.reporter.error(
542                        'Too many autonumbered footnote references: only %s '
543                        'corresponding footnotes available.'
544                        % len(self.autofootnote_labels), base_node=ref)
545                  msgid = self.document.set_id(msg)
546                  for ref in self.document.autofootnote_refs[i:]:
547                      if ref.resolved or ref.hasattr('refname'):
548                          continue
549                      prb = nodes.problematic(
550                            ref.rawsource, ref.rawsource, refid=msgid)
551                      prbid = self.document.set_id(prb)
552                      msg.add_backref(prbid)
553                      ref.replace_self(prb)
554                  break
555              ref += nodes.Text(label)
556              id = self.document.nameids[label]
557              footnote = self.document.ids[id]
558              ref['refid'] = id
559              self.document.note_refid(ref)
560              assert len(ref['ids']) == 1
561              footnote.add_backref(ref['ids'][0])
562              ref.resolved = 1
563              i += 1
564  
565      def symbolize_footnotes(self):
566          """Add symbols indexes to "[*]"-style footnotes and references."""
567          labels = []
568          for footnote in self.document.symbol_footnotes:
569              reps, index = divmod(self.document.symbol_footnote_start,
570                                   len(self.symbols))
571              labeltext = self.symbols[index] * (reps + 1)
572              labels.append(labeltext)
573              footnote.insert(0, nodes.label('', labeltext))
574              self.document.symbol_footnote_start += 1
575              self.document.set_id(footnote)
576          i = 0
577          for ref in self.document.symbol_footnote_refs:
578              try:
579                  ref += nodes.Text(labels[i])
580              except IndexError:
581                  msg = self.document.reporter.error(
582                        'Too many symbol footnote references: only %s '
583                        'corresponding footnotes available.' % len(labels),
584                        base_node=ref)
585                  msgid = self.document.set_id(msg)
586                  for ref in self.document.symbol_footnote_refs[i:]:
587                      if ref.resolved or ref.hasattr('refid'):
588                          continue
589                      prb = nodes.problematic(
590                            ref.rawsource, ref.rawsource, refid=msgid)
591                      prbid = self.document.set_id(prb)
592                      msg.add_backref(prbid)
593                      ref.replace_self(prb)
594                  break
595              footnote = self.document.symbol_footnotes[i]
596              assert len(footnote['ids']) == 1
597              ref['refid'] = footnote['ids'][0]
598              self.document.note_refid(ref)
599              footnote.add_backref(ref['ids'][0])
600              i += 1
601  
602      def resolve_footnotes_and_citations(self):
603          """ 
604          Link manually-labeled footnotes and citations to/from their 
605          references. 
606          """
607          for footnote in self.document.footnotes:
608              for label in footnote['names']:
609                  if label in self.document.footnote_refs:
610                      reflist = self.document.footnote_refs[label]
611                      self.resolve_references(footnote, reflist)
612          for citation in self.document.citations:
613              for label in citation['names']:
614                  if label in self.document.citation_refs:
615                      reflist = self.document.citation_refs[label]
616                      self.resolve_references(citation, reflist)
617  
618      def resolve_references(self, note, reflist):
619          assert len(note['ids']) == 1
620          id = note['ids'][0]
621          for ref in reflist:
622              if ref.resolved:
623                  continue
624              ref.delattr('refname')
625              ref['refid'] = id
626              assert len(ref['ids']) == 1
627              note.add_backref(ref['ids'][0])
628              ref.resolved = 1
629          note.resolved = 1

635  class Substitutions(Transform):
636  
637      """ 
638      Given the following ``document`` as input:: 
639   
640          <document> 
641              <paragraph> 
642                  The 
643                  <substitution_reference refname="biohazard"> 
644                      biohazard 
645                   symbol is deservedly scary-looking. 
646              <substitution_definition name="biohazard"> 
647                  <image alt="biohazard" uri="biohazard.png"> 
648   
649      The ``substitution_reference`` will simply be replaced by the 
650      contents of the corresponding ``substitution_definition``. 
651   
652      The transformed result will be:: 
653   
654          <document> 
655              <paragraph> 
656                  The 
657                  <image alt="biohazard" uri="biohazard.png"> 
658                   symbol is deservedly scary-looking. 
659              <substitution_definition name="biohazard"> 
660                  <image alt="biohazard" uri="biohazard.png"> 
661      """
662  
663      default_priority = 220
664      """The Substitutions transform has to be applied very early, before 
665      `docutils.transforms.frontmatter.DocTitle` and others."""
666  
667      def apply(self):
668          defs = self.document.substitution_defs
669          normed = self.document.substitution_names
670          nested = {}
671          line_length_limit = getattr(self.document.settings,
672                                      "line_length_limit", 10000)
673  
674          subreflist = list(self.document.traverse(nodes.substitution_reference))
675          for ref in subreflist:
676              msg = ''
677              refname = ref['refname']
678              if refname in defs:
679                  key = refname
680              else:
681                  normed_name = refname.lower()
682                  key = normed.get(normed_name, None)
683              if key is None:
684                  msg = self.document.reporter.error(
685                        'Undefined substitution referenced: "%s".'
686                        % refname, base_node=ref)
687              else:
688                  subdef = defs[key]
689                  if len(subdef.astext()) > line_length_limit:
690                      msg = self.document.reporter.error(
691                              'Substitution definition "%s" exceeds the'
692                              ' line-length-limit.' % (key))
693              if msg:
694                  msgid = self.document.set_id(msg)
695                  prb = nodes.problematic(
696                        ref.rawsource, ref.rawsource, refid=msgid)
697                  prbid = self.document.set_id(prb)
698                  msg.add_backref(prbid)
699                  ref.replace_self(prb)
700                  continue
701  
702              parent = ref.parent
703              index = parent.index(ref)
704              if  ('ltrim' in subdef.attributes
705                      or 'trim' in subdef.attributes):
706                  if index > 0 and isinstance(parent[index - 1],
707                                              nodes.Text):
708                      parent[index - 1] = parent[index - 1].rstrip()
709              if  ('rtrim' in subdef.attributes
710                      or 'trim' in subdef.attributes):
711                  if  (len(parent) > index + 1
712                          and isinstance(parent[index + 1], nodes.Text)):
713                      parent[index + 1] = parent[index + 1].lstrip()
714              subdef_copy = subdef.deepcopy()
715              try:
716                  # Take care of nested substitution references:
717                  for nested_ref in subdef_copy.traverse(
718                          nodes.substitution_reference):
719                      nested_name = normed[nested_ref['refname'].lower()]
720                      if nested_name in nested.setdefault(nested_name, []):
721                          raise CircularSubstitutionDefinitionError
722                      nested[nested_name].append(key)
723                      nested_ref['ref-origin'] = ref
724                      subreflist.append(nested_ref)
725              except CircularSubstitutionDefinitionError:
726                  parent = ref.parent
727                  if isinstance(parent, nodes.substitution_definition):
728                      msg = self.document.reporter.error(
729                          'Circular substitution definition detected:',
730                          nodes.literal_block(parent.rawsource,
731                                              parent.rawsource),
732                          line=parent.line, base_node=parent)
733                      parent.replace_self(msg)
734                  else:
735                      # find original ref substitution which caused this error
736                      ref_origin = ref
737                      while ref_origin.hasattr('ref-origin'):
738                          ref_origin = ref_origin['ref-origin']
739                      msg = self.document.reporter.error(
740                          'Circular substitution definition referenced: '
741                          '"%s".' % refname, base_node=ref_origin)
742                      msgid = self.document.set_id(msg)
743                      prb = nodes.problematic(
744                          ref.rawsource, ref.rawsource, refid=msgid)
745                      prbid = self.document.set_id(prb)
746                      msg.add_backref(prbid)
747                      ref.replace_self(prb)
748                  continue
749              ref.replace_self(subdef_copy.children)
750              # register refname of the replacement node(s)
751              # (needed for resolution of references)
752              for node in subdef_copy.children:
753                  if isinstance(node, nodes.Referential):
754                      # HACK: verify refname attribute exists.
755                      # Test with docs/dev/todo.txt, see. |donate|
756                      if 'refname' in node:
757                          self.document.note_refname(node)

760  class TargetNotes(Transform):
761  
762      """ 
763      Creates a footnote for each external target in the text, and corresponding 
764      footnote references after each reference. 
765      """
766  
767      default_priority = 540
768      """The TargetNotes transform has to be applied after `IndirectHyperlinks` 
769      but before `Footnotes`."""
770  
771  
772      def __init__(self, document, startnode):
773          Transform.__init__(self, document, startnode=startnode)
774  
775          self.classes = startnode.details.get('class', [])
776  
777      def apply(self):
778          notes = {}
779          nodelist = []
780          for target in self.document.traverse(nodes.target):
781              # Only external targets.
782              if not target.hasattr('refuri'):
783                  continue
784              names = target['names']
785              refs = []
786              for name in names:
787                  refs.extend(self.document.refnames.get(name, []))
788              if not refs:
789                  continue
790              footnote = self.make_target_footnote(target['refuri'], refs,
791                                                   notes)
792              if target['refuri'] not in notes:
793                  notes[target['refuri']] = footnote
794                  nodelist.append(footnote)
795          # Take care of anonymous references.
796          for ref in self.document.traverse(nodes.reference):
797              if not ref.get('anonymous'):
798                  continue
799              if ref.hasattr('refuri'):
800                  footnote = self.make_target_footnote(ref['refuri'], [ref],
801                                                       notes)
802                  if ref['refuri'] not in notes:
803                      notes[ref['refuri']] = footnote
804                      nodelist.append(footnote)
805          self.startnode.replace_self(nodelist)
806  
807      def make_target_footnote(self, refuri, refs, notes):
808          if refuri in notes:  # duplicate?
809              footnote = notes[refuri]
810              assert len(footnote['names']) == 1
811              footnote_name = footnote['names'][0]
812          else:                           # original
813              footnote = nodes.footnote()
814              footnote_id = self.document.set_id(footnote)
815              # Use uppercase letters and a colon; they can't be
816              # produced inside names by the parser.
817              footnote_name = 'TARGET_NOTE: ' + footnote_id
818              footnote['auto'] = 1
819              footnote['names'] = [footnote_name]
820              footnote_paragraph = nodes.paragraph()
821              footnote_paragraph += nodes.reference('', refuri, refuri=refuri)
822              footnote += footnote_paragraph
823              self.document.note_autofootnote(footnote)
824              self.document.note_explicit_target(footnote, footnote)
825          for ref in refs:
826              if isinstance(ref, nodes.target):
827                  continue
828              refnode = nodes.footnote_reference(refname=footnote_name, auto=1)
829              refnode['classes'] += self.classes
830              self.document.note_autofootnote_ref(refnode)
831              self.document.note_footnote_ref(refnode)
832              index = ref.parent.index(ref) + 1
833              reflist = [refnode]
834              if not utils.get_trim_footnote_ref_space(self.document.settings):
835                  if self.classes:
836                      reflist.insert(0, nodes.inline(text=' ', Classes=self.classes))
837                  else:
838                      reflist.insert(0, nodes.Text(' '))
839              ref.parent.insert(index, reflist)
840          return footnote

843  class DanglingReferences(Transform):
844  
845      """ 
846      Check for dangling references (incl. footnote & citation) and for 
847      unreferenced targets. 
848      """
849  
850      default_priority = 850
851  
852      def apply(self):
853          visitor = DanglingReferencesVisitor(
854              self.document,
855              self.document.transformer.unknown_reference_resolvers)
856          self.document.walk(visitor)
857          # *After* resolving all references, check for unreferenced
858          # targets:
859          for target in self.document.traverse(nodes.target):
860              if not target.referenced:
861                  if target.get('anonymous'):
862                      # If we have unreferenced anonymous targets, there
863                      # is already an error message about anonymous
864                      # hyperlink mismatch; no need to generate another
865                      # message.
866                      continue
867                  if target['names']:
868                      naming = target['names'][0]
869                  elif target['ids']:
870                      naming = target['ids'][0]
871                  else:
872                      # Hack: Propagated targets always have their refid
873                      # attribute set.
874                      naming = target['refid']
875                  self.document.reporter.info(
876                      'Hyperlink target "%s" is not referenced.'
877                      % naming, base_node=target)

880  class DanglingReferencesVisitor(nodes.SparseNodeVisitor):
881  
882      def __init__(self, document, unknown_reference_resolvers):
883          nodes.SparseNodeVisitor.__init__(self, document)
884          self.document = document
885          self.unknown_reference_resolvers = unknown_reference_resolvers
886  
887      def unknown_visit(self, node):
888          pass
889  
890      def visit_reference(self, node):
891          if node.resolved or not node.hasattr('refname'):
892              return
893          refname = node['refname']
894          id = self.document.nameids.get(refname)
895          if id is None:
896              for resolver_function in self.unknown_reference_resolvers:
897                  if resolver_function(node):
898                      break
899              else:
900                  if refname in self.document.nameids:
901                      msg = self.document.reporter.error(
902                          'Duplicate target name, cannot be used as a unique '
903                          'reference: "%s".' % (node['refname']), base_node=node)
904                  else:
905                      msg = self.document.reporter.error(
906                          'Unknown target name: "%s".' % (node['refname']),
907                          base_node=node)
908                  msgid = self.document.set_id(msg)
909                  prb = nodes.problematic(
910                        node.rawsource, node.rawsource, refid=msgid)
911                  try:
912                      prbid = node['ids'][0]
913                  except IndexError:
914                      prbid = self.document.set_id(prb)
915                  msg.add_backref(prbid)
916                  node.replace_self(prb)
917          else:
918              del node['refname']
919              node['refid'] = id
920              self.document.ids[id].note_referenced_by(id=id)
921              node.resolved = 1
922  
923      visit_footnote_reference = visit_citation_reference = visit_reference
