1    ''' 
2    Box layout unit test 
3    ==================== 
4     
5    Order matter. 
6    On the screen, most of example must have the red->blue->green order. 
7    '''
8    
9    from kivy.tests.common import GraphicUnitTest
10   
11   
12   class UIXBoxLayoutTestcase(GraphicUnitTest):
13   
14       def box(self, r, g, b):
15           from kivy.uix.widget import Widget
16           from kivy.graphics import Color, Rectangle
17           wid = Widget()
18           with wid.canvas:
19               Color(r, g, b)
20               r = Rectangle(pos=wid.pos, size=wid.size)
21   
22           def linksp(instance, *largs):
23               r.pos = instance.pos
24               r.size = instance.size
25           wid.bind(pos=linksp, size=linksp)
26           return wid
27   
28       def test_boxlayout_orientation(self):
29           from kivy.uix.boxlayout import BoxLayout
30           r = self.render
31           b = self.box
32   
33           layout = BoxLayout()
34           layout.add_widget(b(1, 0, 0))
35           layout.add_widget(b(0, 1, 0))
36           layout.add_widget(b(0, 0, 1))
37           r(layout)
38   
39           layout = BoxLayout(orientation='vertical')
40           layout.add_widget(b(1, 0, 0))
41           layout.add_widget(b(0, 1, 0))
42           layout.add_widget(b(0, 0, 1))
43           r(layout)
44
